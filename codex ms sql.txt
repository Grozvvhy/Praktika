CREATE TABLE statuses (
    id INT NOT NULL IDENTITY(1,1),
    name VARCHAR(50) NOT NULL,
    CONSTRAINT statuses_pkey PRIMARY KEY (id),
    CONSTRAINT statuses_name_key UNIQUE (name)
);
GO

CREATE TABLE departments (
    id INT NOT NULL IDENTITY(1,1),
    name VARCHAR(100) NOT NULL,
    CONSTRAINT departments_pkey PRIMARY KEY (id),
    CONSTRAINT departments_name_key UNIQUE (name)
);
GO

CREATE TABLE department_employees (
    id INT NOT NULL IDENTITY(1,1),
    full_name VARCHAR(150) NOT NULL,
    department_id INT NOT NULL,
    CONSTRAINT department_employees_pkey PRIMARY KEY (id)
);
GO

CREATE TABLE users (
    id INT NOT NULL IDENTITY(1,1),
    email VARCHAR(255) NOT NULL,
    password_hash CHAR(32) NOT NULL,
    registered_at DATETIME DEFAULT GETDATE(),
    CONSTRAINT users_pkey PRIMARY KEY (id),
    CONSTRAINT users_email_key UNIQUE (email)
);
GO

CREATE TABLE visitors (
    id INT NOT NULL IDENTITY(1,1),
    last_name VARCHAR(50) NOT NULL,
    first_name VARCHAR(50) NOT NULL,
    middle_name VARCHAR(50) NULL,
    phone VARCHAR(20) NULL,
    email VARCHAR(255) NOT NULL,
    organization VARCHAR(200) NULL,
    note TEXT NOT NULL,
    birth_date DATE NOT NULL,
    passport_series CHAR(4) NOT NULL,
    passport_number CHAR(6) NOT NULL,
    photo_path VARCHAR(500) NULL,
    passport_scan_path VARCHAR(500) NOT NULL,
    CONSTRAINT visitors_pkey PRIMARY KEY (id),
    CONSTRAINT check_age CHECK (DATEDIFF(YEAR, birth_date, GETDATE()) >= 16),
    CONSTRAINT check_passport_series CHECK (passport_series LIKE '[0-9][0-9][0-9][0-9]'),
    CONSTRAINT check_passport_number CHECK (passport_number LIKE '[0-9][0-9][0-9][0-9][0-9][0-9]')
);
GO

CREATE TABLE requests (
    id INT NOT NULL IDENTITY(1,1),
    user_id INT NOT NULL,
    type VARCHAR(20) NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    purpose TEXT NOT NULL,
    department_id INT NOT NULL,
    employee_id INT NOT NULL,
    status_id INT NOT NULL DEFAULT 1,
    rejection_reason TEXT NULL,
    created_at DATETIME DEFAULT GETDATE(),
    CONSTRAINT requests_pkey PRIMARY KEY (id),
    CONSTRAINT requests_type_check CHECK (type IN ('personal', 'group'))
);
GO

CREATE TABLE request_visitors (
    id INT NOT NULL IDENTITY(1,1),
    request_id INT NOT NULL,
    visitor_id INT NOT NULL,
    order_number INT NOT NULL,
    CONSTRAINT request_visitors_pkey PRIMARY KEY (id),
    CONSTRAINT request_visitors_request_id_order_number_key UNIQUE (request_id, order_number)
);
GO

ALTER TABLE department_employees ADD CONSTRAINT department_employees_department_id_fkey FOREIGN KEY (department_id) REFERENCES departments(id) ON DELETE NO ACTION;
GO

ALTER TABLE requests ADD CONSTRAINT requests_user_id_fkey FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE;
GO

ALTER TABLE requests ADD CONSTRAINT requests_department_id_fkey FOREIGN KEY (department_id) REFERENCES departments(id);
GO

ALTER TABLE requests ADD CONSTRAINT requests_employee_id_fkey FOREIGN KEY (employee_id) REFERENCES department_employees(id);
GO

ALTER TABLE requests ADD CONSTRAINT requests_status_id_fkey FOREIGN KEY (status_id) REFERENCES statuses(id);
GO

ALTER TABLE request_visitors ADD CONSTRAINT request_visitors_request_id_fkey FOREIGN KEY (request_id) REFERENCES requests(id) ON DELETE CASCADE;
GO

ALTER TABLE request_visitors ADD CONSTRAINT request_visitors_visitor_id_fkey FOREIGN KEY (visitor_id) REFERENCES visitors(id) ON DELETE CASCADE;
GO
